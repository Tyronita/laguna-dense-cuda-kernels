#include <torch/extension.h>
#include <cuda_runtime.h>

// Custom CUDA kernel for 2D convolution with stride=1, padding=0, dilation=1
// Assumes input shape: [N, C_in, H, W] and output shape: [N, C_in, H_out, W_out]
// Convolution with kernel size (kH, 1) and stride=1, padding=0, dilation=1.

__global__ void conv2d_stride_kernel(
    const float* input,
    const float* weight,
    const float* bias,
    float* output,
    int N, int C_in, int H, int W,
    int kH, int stride, int padding, int dilation) {

    // Compute output dimensions
    int H_out = H - kH + padding;
    int W_out = W - 1 + padding;

    // Each block processes one (n, c) slice
    int slice_idx = blockIdx.x;  // slice index for (n, c)
    int n = slice_idx / C_in;
    int c = slice_idx % C_in;

    // Each thread processes one output element
    int h_out = blockIdx.y * blockDim.y + threadIdx.y;
    int w_out = blockIdx.x * blockDim.x + threadIdx.x;

    if (h_out < H_out && w_out < W_out) {
        float sum = bias[c];
        // Loop over input channels and kernel height
        for (int c_in = 0; c_in < C_in; c_in++) {
            for (int kh = 0; kh < kH; kh++) {
                int h_in = h_out - padding + kh - dilation;
                if (h_in >= 0 && h_in < H) {
                    for (int iw = 0; iw < W; iw++) {
                        int input_idx = n * (C_in * H * W) + c_in * (H * W) + h_in * W + iw;
                        int weight_idx = c_in * (kH * 1) + kh * 1;
                        sum += input[input_idx] * weight[weight_idx];
                    }
                }
            }
        }
        int output_idx = n * (C_in * H_out * W_out) + c * (H_out * W_out) + h_out * W_out + w_out;
        output[output_idx] = sum;
    }
}

// The forward function exposed via PyBind11.
// Input:
//   x: Tensor of shape [N, C_in, H, W] (CUDA)
//   weight: Tensor of shape [C_in, C_in, kH, 1] (CUDA)
//   bias: Tensor of shape [C_in] (CUDA) if applicable.

// Output:
//   out: Tensor of shape [N, C_in, H_out, W_out] computed by conv2d with stride=1, padding=0, dilation=1.

torch::Tensor forward(
    torch::Tensor x,
    torch::Tensor weight,
    torch::Tensor bias,
    int64_t stride,
    int64_t padding,
    int64_t dilation) {

    // Ensure inputs are on CUDA
    TORCH_CHECK(x.is_cuda(), "x must be a CUDA tensor");
    TORCH_CHECK(weight.is_cuda(), "weight must be a CUDA tensor");
    TORCH_CHECK(bias.is_cuda(), "bias must be a CUDA tensor");

    // Get dimensions
    int N = x.size(0);
    int C_in = x.size(1);
    int H = x.size(2);
    int W = x.size(3);
    int kH = weight.size(2);

    // Compute output dimensions
    int H_out = H - kH + padding;
    int W_out = W - 1 + padding;

    // Allocate output tensor
    auto out = torch::empty({N, C_in, H_out, W_out}, x.options());

    // Launch one block per (n, c) slice
    dim3 gridDim1(N * C_in);
    dim3 blockDim1(16, 16);  // 256 threads per block

    // Calculate grid dimensions for output spatial
    dim3 gridDim2((H_out + blockDim.y - 1) / blockDim.y, (W_out + blockDim.x - 1) / blockDim.x);

    conv2d_stride_kernel<<<gridDim1, gridDim2>>>(
        x.data_ptr<float>(),
        weight.data_ptr<float>(),
        bias.data_ptr<float>(),
        out.data_ptr<float>(),
        N, C_in, H, W,
        kH, stride, padding, dilation
    );

    cudaError_t err = cudaGetLastError();
    TORCH_CHECK(err == cudaSuccess, "CUDA kernel failed: ", cudaGetErrorString(err));

    return out;
}

PYBIND11_MODULE(TORCH_EXTENSION_NAME, m) {
    m.def("forward", &forward, "2D convolution with stride=1, padding=0, dilation=1 (CUDA)");
}
