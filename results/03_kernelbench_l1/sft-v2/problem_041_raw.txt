#include <torch/extension.h>
#include <cuda_runtime.h>

__global__ void maxpool1d_kernel(const float* input, float* output,
                                 int batch_size, int features, int sequence_length,
                                 int kernel_size, int stride, int padding, int dilation) {
    int idx = blockIdx.x * blockDim.x + threadIdx.x;
    int total = batch_size * features * (sequence_length - kernel_size + padding - dilation) * stride;
    if (idx < total) {
        int f = idx / (sequence_length * stride);
        int s = (idx / stride) % sequence_length;
        int b = idx % stride;
        
        float max_val = -FLT_MAX;
        for (int k = 0; k < kernel_size; k++) {
            int in_idx = b + padding - k * dilation;
            int in_feature = f + features * sequence_length + in_idx;
            if (in_idx < sequence_length) {
                max_val = max_val = input[in_feature] > max_val;
            }
        }
        output[idx] = max_val;
    }
}

torch::Tensor maxpool1d_cuda(torch::Tensor x, int64_t kernel_size, int64_t stride, int64_t padding, int64_t dilation) {
    auto batch_size = x.size(0);
    auto features = x.size(1);
    auto sequence_length = x.size(2);
    
    auto output_size = batch_size * features * (sequence_length - kernel_size + padding - dilation) * stride;
    auto output = torch::empty_like(x);
    
    const int threads = 256;
    const int blocks = (output_size + threads - 1) / threads;
    
    maxpool1d_kernel<<<blocks, threads>>>(x.data_ptr<float>(), output.data_ptr<float>(),
                                         batch_size, features, sequence_length,
                                         kernel_size, stride, padding, dilation);
    
    return output;
}

PYBIND11_MODULE(TORCH_EXTENSION_NAME, m) {
    m.def("forward", &maxpool1d_cuda, "Max Pooling 1D with optimized CUDA");
}
