#include <torch/extension.h>

// Custom CUDA kernel for fused elementwise addition and ReLU activation
__global__ void fused_add_relu_kernel(const float* a, const float* b, float* out, int size) {
    int idx = blockIdx.x * blockDim.x + threadIdx.x;
    if (idx < size) {
        float sum = a[idx] + b[idx];
        out[idx] = sum > 0.0f ? sum : 0.0f;
    }
}

// Custom CUDA kernel for scaled dot product attention
__global__ void scaled_dot_product_attention_kernel(
    const float* __restrict__ Q,

    const float* __restrict__ V,
    float* __restrict__ out,
    int batch_size,
    int num_heads,
    int sequence_length,
    int embedding_dim) {

    int idx = blockIdx.x * blockDim.x + threadIdx.x;
    if (idx < batch_size * num_heads * sequence_length) {
        int b = idx / (num_heads * sequence_length);
        int h = (idx / sequence_length) % num_heads;
        int s = idx % sequence_length;

        float sum = 0.0f;
        for (int e = 0; e < embedding_dim; e++) {
            int q_idx = b * (num_heads * embedding_dim) + h * embedding_dim + e;
            int k_idx = h * (num_heads * embedding_dim) + e;
            sum += Q[q_idx] * K[k_idx];
        }
        sum = sum / embedding_dim;
        out[idx] = sum * V[b * (num_heads * embedding_dim) + h * embedding_dim + s];
    }
}
