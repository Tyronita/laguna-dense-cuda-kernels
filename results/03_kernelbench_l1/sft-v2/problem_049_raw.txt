#include <torch/extension.h>
#include <cuda_runtime.h>

__global__ void maxReductionKernel(const float* input, float* output, int batch_size, int dim1, int dim2) {
    extern __shared__ float shared_mem[];
    int tid = threadIdx.x;
    int bid = blockIdx.x;
    int batch = bid / dim1;
    int dim_idx = bid % dim1;
    
    float max_val = -FLT_MAX;
    
    // Each thread processes multiple elements in the shared memory
    for (int i = tid; i < dim2; i += blockDim.x) {
        int idx = batch * dim1 * dim2 + dim_idx * dim2 + i;
        max_val = max(max_val, input[idx]);
    }
    
    // Store results in shared memory
    shared_mem[tid] = max_val;
    __syncthreads();
    
    // Reduce within the dimension using shared memory
    for (int s = blockDim.x / 2; s > 0; s >>= 1) {
        if (tid < s) {
            shared_mem[tid] = max(shared_mem[tid], shared_mem[tid + s]);
        }
        __syncthreads();
    }
    
    // Thread 0 writes the final result
    if (tid == 0) {
        output[batch * dim1 + dim_idx] = shared_mem[0];
    }
}

torch::Tensor maxReduction_cuda(torch::Tensor x) {
    TORCH_CHECK(x.dim() == 3, "Input must be 3D");
    TORCH_CHECK(x.size(1) == x.size(2), "Dimension mismatch");
    TORCH_CHECK(x.is_cuda() && x.is_contiguous(), "Input must be on CUDA");
    
    int batch_size = x.size(0);
    int dim1 = x.size(1);
    int dim2 = x.size(2);
    
    auto output = torch::zeros({batch_size, dim1}, x.options());
    
    const int threads = 256;
    const int blocks = batch_size * dim1;
    const int shared_mem_size = threads * sizeof(float);
    
    maxReductionKernel<<<blocks, threads, shared_mem_size>>>(
        x.data_ptr<float>(),
        output.data_ptr<float>(),
        batch_size,
        dim1,
        dim2
    );
    
    return output;
}

PYBIND11_MODULE(TORCH_EXTENSION_NAME, m) {
    m.def("forward", &maxReduction_cuda, "Max reduction over specified dimension (CUDA)");
}
