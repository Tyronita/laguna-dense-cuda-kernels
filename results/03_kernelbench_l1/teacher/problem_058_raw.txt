import torch
import torch.nn as nn
import torch.nn.functional as F
from torch.utils.cpp_extension import load_inline

conv_transpose3d_source = """
#include <torch/extension.h>
#include <cuda_runtime.h>

template<typename T>
__device__ void load_input_block(const T* input, int n, T* block, int stride_batch, int stride_depth, int stride_height, int stride_width) {
    int idx = blockIdx.x * blockDim.x + threadIdx.x;
    if (idx < n) {
        block[idx] = input[idx];
    }
}

template<typename T>
__global__ void conv_transpose3d_kernel(
    const T* input,
    const T* weight,
    T* output,
    int batch_size, int in_channels, int depth_in, int height_in, int width_in,
    int out_channels, int depth_out, int height_out, int width_out,
    int kernel_d, int kernel_h, int kernel_w,
    int stride_d, int stride_h, int stride_w,
    int pad_d, int pad_h, int pad_w,
    int out_pad_d, int out_pad_h, int out_pad_w
) {
    int idx = blockIdx.x * blockDim.x + threadIdx.x;
    int total_elements = batch_size * out_channels * depth_out * height_out * width_out;
    
    if (idx >= total_elements) return;
    
    int temp = idx;
    int wid = temp % width_out;
    temp /= width_out;
    int hid = temp % height_out;
    temp /= height_out;
    int did = temp % depth_out;
    temp /= depth_out;
    int oid = temp % out_channels;
    temp /= out_channels;
    int bid = temp;
    
    int d_in = did * stride_d - pad_d;
    int h_in = hid * stride_h - pad_h;
    int w_in = wid * stride_w - pad_w;
    
    T sum = 0;
    for (int kd = 0; kd < kernel_d; kd++) {
        for (int kh = 0; kh < kernel_h; kh++) {
            for (int kw = 0; kw < kernel_w; kw++) {
                int d_curr = d_in + kd;
                int h_curr = h_in + kh;
                int w_curr = w_in + kw;
                
                if (d_curr >= 0 && d_curr < depth_in && h_curr >= 0 && h_curr < height_in && w_curr >= 0 && w_curr < width_in) {
                    for (int ic = 0; ic < in_channels; ic++) {
                        int weight_idx = ((kd * kernel_h + kh) * kernel_w + kw) * in_channels + ic;
                        int input_idx = ((bid * in_channels + ic) * depth_in + d_curr) * height_in + h_curr) * width_in + w_curr;
                        sum += input[input_idx] * weight[weight_idx];
                    }
                }
            }
        }
    }
    
    int out_idx = ((bid * out_channels + oid) * depth_out + did) * height_out + hid) * width_out + wid;
    output[out_idx] = sum;
}

torch::Tensor conv_transpose3d_cuda(
    torch::Tensor input,
    torch::Tensor weight,
    torch::Tensor bias,
    int stride_d, int stride_h, int stride_w,
    int pad_d, int pad_h, int pad_w,
    int out_pad_d, int out_pad_h, int out_pad_w
) {
    auto batch_size = input.size(0);
    auto in_channels = input.size(1);
    auto depth_in = input.size(2);
    auto height_in = input.size(3);
    auto width_in = input.size(4);
    
    auto kernel_d = weight.size(2);
    auto kernel_h = weight.size(3);
    auto kernel_w = weight.size(4);
    
    auto depth_out = (depth_in - 1) * stride_d - 2 * pad_d + kernel_d + out_pad_d;
    auto height_out = (height_in - 1) * stride_h - 2 * pad_h + kernel_h + out_pad_h;
    auto width_out = (width_in - 1) * stride_w - 2 * pad_w + kernel_w + out_pad_w;
    auto out_channels = weight.size(0);
    
    auto output = torch::zeros({batch_size, out_channels, depth_out, height_out, width_out}, input.options());
    
    const int block_size = 256;
    const int num_blocks = (batch_size * out_channels * depth_out * height_out * width_out + block_size - 1) / block_size;
    
    conv_transpose3d_kernel<<<num_blocks, block_size>>>(
        input.data_ptr<float>(),
        weight.data_ptr<float>(),
        output.data_ptr<float>(),
        batch_size, in_channels, depth_in, height_in, width_in,
        out_channels, depth_out, height_out, width_out,
        kernel_d, kernel_h, kernel_w,
        stride_d, stride_h, stride_w,
        pad_d, pad_h, pad_w,
        out_pad_d, out_pad_h, out_pad_w
    );
    
    if (bias.defined()) {
        output = output + bias.view({1, out_channels, 1, 1, 1});
    }
    
    return output;
}
"""

conv_transpose3d_cpp_source = """
torch::Tensor conv_transpose3d_cuda(
    torch::Tensor input,
    torch::Tensor weight,
    torch::Tensor bias,
    int stride_d, int stride_h, int stride_w,
    int pad_d, int pad_h, int pad_w,

conv_transpose3d = load_inline(
    name="conv_transpose3d",
    cpp_sources=conv_transpose3d_cpp_source,
    cuda_sources=conv_transpose3d_source,
    functions=["conv_transpose3d_cuda"],
    verbose=True
)

class ModelNew(nn.Module):
    def __init__(self, in_channels: int, out_channels: int, kernel_size: tuple, stride: tuple = (1, 1, 1), padding: tuple = (0, 0, 0), output_padding: tuple = (0, 0, 0), groups: int = 1, bias: bool = False):
        super(ModelNew, self).__init__()
        self.in_channels = in_channels
        self.out_channels = out_channels
        self.kernel_size = kernel_size
        self.stride = stride
        self.padding = padding
        self.output_padding = output_padding
        self.groups = groups
        self.bias = bias
        
        self.weight = nn.Parameter(torch.randn(out_channels, in_channels // groups, *kernel_size))
        if bias:
            self.bias_tensor = nn.Parameter(torch.randn(out_channels))
        else:
            self.bias_tensor = None
        
        self.conv_transpose3d = conv_transpose3d

    def forward(self, x: torch.Tensor) -> torch.Tensor:
        return self.conv_transpose3d.conv_transpose3d_cuda(
            x,
            self.weight,
            self.bias_tensor if self.bias else torch.Tensor(),
            self.stride[0], self.stride[1], self.stride[2],
            self.padding[0], self.padding[1], self.padding[2],
            self.output_padding[0], self.output_padding[1], self.output_padding[2]
        )
