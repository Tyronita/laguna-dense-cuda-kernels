import torch
import torch.nn as nn
from torch.utils.cpp_extension import load_inline

conv2d_source = """
#include <torch/extension.h>
#include <cuda_runtime.h>

template<typename T>
__device__ void load_tile(const T* __restrict__ input, T tile[3][3], int h, int w, int C, int H, int W, int c_start, int dilation_h, int dilation_w) {
    for (int dh = 0; dh < 3; dh++) {
        for (int dw = 0; dw < 3; dw++) {
            int nh = h * dilation_h + dh * dilation_h;
            int nw = w * dilation_w + dw * dilation_w;
            if (nh >= 0 && nh < H && nw >= 0 && nw < W) {
                tile[dh][dw] = input[nh * W * C + nw * C + c_start];
            } else {
                tile[dh][dw] = 0;
            }
        }
    }
}

template<typename T>
__global__ void conv2d_kernel(const T* __restrict__ input, const T* __restrict__ weight, const T* __restrict__ bias, T* __restrict__ output,
                              int N, int C, int H, int W, int K_h, int K_w, int S_h, int S_w, int P_h, int P_w, int D_h, int D_w, int C_out, bool has_bias) {
    int h_out = blockIdx.x;
    int w_out = blockIdx.y;
    int c_out = blockIdx.z;
    int n = blockIdx.w;
    
    if (h_out >= (H + 2*P_h - D_h*(K_h-1) - 1)/S_h + 1 || w_out >= (W + 2*P_w - D_w*(K_w-1) - 1)/S_w + 1) return;
    
    int h_in_start = h_out * S_h - P_h;
    int w_in_start = w_out * S_w - P_w;
    
    T sum = 0;
    
    for (int c_in = 0; c_in < C; c_in++) {
        T tile[3][3];
        load_tile(input + n * C * H * W, tile, h_in_start, w_in_start, C, H, W, c_in, D_h, D_w);
        
        int w_idx = c_out * C + c_in;
        for (int kh = 0; kh < K_h; kh++) {
            for (int kw = 0; kw < K_w; kw++) {
                sum += tile[kh][kw] * weight[w_idx * K_h * K_w + kh * K_w + kw];
            }
        }
    }
    
    if (has_bias) {
        sum += bias[c_out];
    }
    
    output[n * C_out * ((H + 2*P_h - D_h*(K_h-1) - 1)/S_h + 1) * ((W + 2*P_w - D_w*(K_w-1) - 1)/S_w + 1) + h_out * ((W + 2*P_w - D_w*(K_w-1) - 1)/S_w + 1) + w_out] = sum;
}

torch::Tensor conv2d_cuda(torch::Tensor input, torch::Tensor weight, torch::Tensor bias, int stride_h, int stride_w, int padding_h, int padding_w, int dilation_h, int dilation_w) {
    int N = input.size(0);
    int C = input.size(1);
    int H = input.size(2);
    int W = input.size(3);
    int K_h = weight.size(2);
    int K_w = weight.size(3);
    int C_out = weight.size(0);
    
    int H_out = (H + 2 * padding_h - dilation_h * (K_h - 1) - 1) / stride_h + 1;
    int W_out = (W + 2 * padding_w - dilation_w * (K_w - 1) - 1) / stride_w + 1;
    
    auto output = torch::zeros({N, C_out, H_out, W_out}, input.options());
    
    if (H_out <= 0 || W_out <= 0) return output;
    
    dim3 block(8, 8, 8, 1);
    dim3 grid(H_out, W_out, C_out, N);
    
    conv2d_kernel<<<grid, block>>>(
        input.data_ptr<float>(),
        weight.data_ptr<float>(),
        bias.defined() ? bias.data_ptr<float>() : nullptr,
        output.data_ptr<float>(),
        N, C, H, W, K_h, K_w, stride_h, stride_w, padding_h, padding_w, dilation_h, dilation_w, C_out, bias.defined()
    );
    
    return output;
}
"""

conv2d_cpp_source = """
torch::Tensor conv2d_cuda(torch::Tensor input, torch::Tensor weight, torch::Tensor bias, int stride_h, int stride_w, int padding_h, int padding_w, int dilation_h, int dilation_w);
"""

conv2d = load_inline(
    name="conv2d",
    cpp_sources=conv2d_cpp_source,
    cuda_sources=conv2d_source,
    functions=["conv2d_cuda"],
    verbose=True
)

class ModelNew(nn.Module):
    def __init__(self, in_channels: int, out_channels: int, kernel_size: tuple, stride: tuple = (1, 1), padding: tuple = (0, 0), dilation: tuple = (1, 1), groups: int = 1, bias: bool = False):
        super(ModelNew, self).__init__()
        self.in_channels = in_channels
        self.out_channels = out_channels
        self.kernel_size = kernel_size
        self.stride = stride
        self.padding = padding
        self.dilation = dilation
        self.groups = groups
        self.bias = bias
        
        weight = torch.randn(out_channels, in_channels // groups, *kernel_size)
        self.weight = torch.nn.Parameter(weight)
        
        if bias:
            self.bias = torch.nn.Parameter(torch.randn(out_channels))
        else:
            self.bias = None
        
        self.conv2d = conv2d

    def forward(self, x: torch.Tensor) -> torch.Tensor:
        return self.conv2d.conv2d_cuda(
            x, 
            self.weight, 
            self.bias if self.bias is not None else torch.Tensor(), 
            self.stride[0], 
            self.stride[1], 
            self.padding[0], 
            self.padding[1], 
            self.dilation[0], 
            self.dilation[1]
        )
