#include <torch/extension.h>
#include <cuda_runtime.h>

__global__ void reshape_kernel(const half* input, half* output, int batch_size, int num_heads, int seq_len, int embed_dim) {
    int idx = blockIdx.x * blockDim.x + threadIdx.x;
    int total_size = batch_size * num_heads * seq_len * embed_dim;
    if (idx < total_size) {
        int original_idx = idx;
        int new_idx = idx;
        output[new_idx] = input[original_idx];
    }
}

__global__ void matmul_kernel(const half* A, const half* B, half* C, int M, int N, int K) {
    int row = blockIdx.y * blockDim.y + threadIdx.y;
    int col = blockIdx.x * blockDim.x + threadIdx.x;
    if (row < M && col < N) {
        half sum = 0;
        for (int k = 0; k < K; k++) {
            sum += A[row * K + k] * B[k * N + col];
        }
        C[row * N + col] = sum;
    }
}

__global__ void softmax_kernel(const half* input, half* output, int batch_size, int num_heads, int seq_len) {
    int idx = blockIdx.x * blockDim.x + threadIdx.x;
    int total_elements = batch_size * num_heads * seq_len;
    if (idx < total_elements) {
        int batch_head = idx / seq_len;
        int pos = idx % seq_len;
        int base = batch_head * seq_len;
        
        half max_val = -1e10;
        for (int i = 0; i < seq_len; i++) {
            max_val = fmaxf(max_val, input[base + i]);
        }
        
        half sum = 0;
        for (int i = 0; i < seq_len; i++) {
            sum += __expf(input[base + i] - max_val);
        }
        
        output[idx] = __expf(input[base + pos] - max_val) / sum;
    }
}

__global__ void dropout_kernel(const half* input, half* output, float prob, unsigned long seed, int size) {
    int idx = blockIdx.x * blockDim.x + threadIdx.x;
    if (idx < size) {
        float rand_val = (float)(rand() % 1000000) / 1000000.0f;
        output[idx] = (rand_val > prob) ? input[idx] / (1.0f - prob) : 0.0f;
    }
}

__global__ void add_kernel(const half* Q, const half* K, const half* V, const half* attn_weights, half* out, int batch_size, int num_heads, int seq_len, int embed_dim) {
    int idx = blockIdx.x * blockDim.x + threadIdx.x;
    int total_size = batch_size * num_heads * seq_len * embed_dim;
    if (idx < total_size) {
        int pos = idx / embed_dim;
        int dim = idx % embed_dim;
        out[idx] = Q[idx] + attn_weights[pos] * (V[idx] - Q[idx]);
    }
}

torch::Tensor scaled_dot_product_attention_cuda(torch::Tensor Q, torch::Tensor K, torch::Tensor V) {
    auto batch_size = Q.size(0);
    auto num_heads = Q.size(1);
    auto seq_len = Q.size(2);
    auto embed_dim = Q.size(3);
    
    auto scale = 1.0f / sqrtf((float)embed_dim);
    
    auto Q_flat = Q.reshape({batch_size * num_heads, seq_len, embed_dim});
    auto K_flat = K.reshape({batch_size * num_heads, seq_len, embed_dim});
    auto V_flat = V.reshape({batch_size * num_heads, seq_len, embed_dim});
    
    auto attn_weights = torch::zeros({batch_size * num_heads, seq_len, seq_len}, torch::dtype(torch::kHalf).device(Q.device()));
    
    const int block_size = 16;
    dim3 grid((embed_dim + block_size - 1) / block_size, seq_len, batch_size * num_heads);
    matmul_kernel<<<grid, dim3(block_size, block_size), 0, 0>>>(
        Q_flat.data_ptr<at::Half>(), 
        K_flat.data_ptr<at::Half>(), 
        attn_weights.data_ptr<at::Half>(), 
        batch_size * num_heads, seq_len, embed_dim
    );
    
    for (int i = 0; i < batch_size * num_heads; i++) {
        softmax_kernel<<<1, seq_len, 0, 0>>>(
            attn_weights.data_ptr<at::Half>() + i * seq_len,
            attn_weights.data_ptr<at::Half>() + i * seq_len,
            1, 1, seq_len
        );
    }
    
    auto out = torch::zeros_like(Q);
    add_kernel<<<grid, dim3(block_size, block_size), 0, 0>>>(
        Q_flat.data_ptr<at::Half>(),
        K_flat.data_ptr<at::Half>(),
        V_flat.data_ptr<at::Half>(),
        attn_weights.data_ptr<at::Half>(),
        out.data_ptr<at::Half>(),
        batch_size, num_heads, seq_len, embed_dim
    );
    
    return out;
}
"""
