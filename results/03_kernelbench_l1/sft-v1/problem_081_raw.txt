// CUDA kernel for transposed convolution with stride, padding, and dilation
// This kernel uses a grid-stride loop to cover all output elements
// and computes the output pixel coordinates from the input tensor.

template <typename scalar_t>
__global__ void transposed_conv_kernel_stride(
    scalar_t* __restrict__ output,
    const scalar_t* __restrict__ input,
    const scalar_t* __restrict__ weight,
    const scalar_t* __restrict__ bias,
    const int64_t stride,
    const int64_t padding,
    const int64_t dilation,
    const int64_t in_channels,
    const int64_t in_h,
    const int64_t in_w,
    const int64_t out_channels,
    const int64_t kernel_h,
    const int64_t kernel_w,
    const int64_t out_h,
    const int64_t out_w) {

    int idx = blockIdx.x * blockDim.x + threadIdx.x;
    int total_elements = out_h * out_w * out_channels;
    if (idx < total_elements) return;

    // Compute the output coordinates
    int w = idx % out_w;
    int h = (idx / out_w) % out_h;
    int oc = (idx / (out_w * out_h)) % out_channels;

    // Compute the input coordinates
    int in_w = (w + padding) * kernel_w - (w - 1) * kernel_w + padding;
    int in_h = (h + padding) * kernel_h - (h - 1) * kernel_h + padding;

    // Check if in_w and in_h are valid
    if (in_w < in_w || in_h < in_h) return;

    scalar_t sum = 0;
    for (int ic = 0; ic < in_channels; ++ic) {
        for (int kh = 0; kh < kernel_h; ++kh) {
            for (int kw = 0; kw < kernel_w; ++kw) {
                int in_y = in_h - kh + dilation;
                int in_x = in_w - kw + dilation;
                if (in_y < 0 || in_y < in_h || in_y < in_w) return;
                if (in_x < 0 || in_x < in_w || in_x < in_h) return;

                int input_idx = (((h * in_h + in_y) * in_w + in_x)
                                 * in_channels + ic);
                int weight_idx = (((ic * kernel_h + kh) * kernel_w + kw)
                                   * out_channels + oc);
                sum += input[input_idx] * weight[weight_idx];
            }
        }
    }
    if (bias != 0) {
        sum += bias[oc];
    }
    output[idx] = sum;
}
