#include <torch/extension.h>
#include <cuda.h>
#include <cuda_runtime.h>

// CUDA kernel for elementwise addition
__global__ void elementwise_add_kernel(const float* __restrict__ a,
                                        const float* __restrict__ b,
                                        float* __restrict__ out,
                                        int size) {
    int idx = blockIdx.x * blockDim.x + threadIdx.x;
    if (idx < size) {
        out[idx] = a[idx] + b[idx];
    }
}

// PyTorch wrapper function for elementwise addition
torch::Tensor elementwise_add_cuda(torch::Tensor a, torch::Tensor b) {
    auto size = a.numel();
    auto out = torch::zeros_like(a);
    const int block_size = 256;
    const int num_blocks = (size + block_size - 1) / block_size;
    elementwise_add_kernel<<<num_blocks, block_size>>>(
        a.data_ptr<float>(),
        b.data_ptr<float>(),
        out.data_ptr<float>(),
        size
    );
    return out;
}

// PyTorch wrapper function for cumulative sum along a specified dimension
torch::Tensor cumsum_cuda(torch::Tensor x, int dim) {
    auto out = torch::empty_like(x);
    const int block_size = 256;
    const int num_blocks = (x.numel() + block_size - 1) / block_size;
    elementwise_add_kernel<<<num_blocks, block_size>>>(
        x.data_ptr<float>(),
        out.data_ptr<float>(),
        out.data_ptr<float>(),
        x.numel()
    );
    return out.sum(out, dim);
}
