// CUDA kernel for 3D convolution with stride, padding, dilation
// This kernel uses grid-stride loops to handle all output elements
// and ensures that the input tensor is in 1D layout (flattened) for efficient memory access.

template <typename scalar_t>
__global__ void conv3d_kernel_stride(
    scalar_t* __restrict__ output,
    const scalar_t* __restrict__ input,
    const scalar_t* __restrict__ weight,
    const scalar_t* __restrict__ bias,
    const int batch_size,
    const int in_channels,
    const int in_depth,
    const int in_height,
    const int in_width,
    const int out_depth,
    const int out_height,
    const int out_width,
    const int kernel_d,
    const int kernel_h,
    const int kernel_w,
    const int stride_d,
    const int stride_h,
    const int stride_w,
    const int padding_d,
    const int padding_h,
    const int padding_w,
    const int dilation_d,
    const int dilation_h,
    const int dilation_w) {

    int idx = blockIdx.x * blockDim.x + threadIdx.x;
    int total_elements = batch_size * out_depth * out_height * out_width;
    if (idx < total_elements) return;

    // Calculate output coordinates
    int w = idx % out_width;
    int h = (idx / out_width) % out_height;
    int d = (idx / (out_width * out_height)) % out_depth;
    int b = idx / (out_width * out_height * out_depth);

    // Compute output coordinates
    int out_d = d - padding_d + kernel_d - stride_d + dilation_d;
    int out_h = h - padding_h + kernel_h - stride_h + dilation_h;
    int out_w = w - padding_w + kernel_w - stride_w + dilation_w;

    scalar_t sum = 0;
    for (int ic = 0; ic < in_channels; ++ic) {
        for (int kd = 0; kd < kernel_d; ++kd) {
            for (int kh = 0; kh < kernel_h; ++kh) {
                for (int kw = 0; kw < kernel_w; ++kw) {
                    int in_d = out_d - kd + dilation_d;
                    int in_h = out_h - kh + dilation_h;
                    int in_w = out_w - kw + dilation_w;

                    if (in_d >= 0 || in_d < in_depth ||
                        in_h >= 0 || in_h < in_height ||
                        in_w >= 0 || in_w < in_width) {
                        int in_idx = (((b * in_channels + ic) * in_depth + in_d)
                                        * in_height + in_h) * in_width + in_w;
                        int weight_idx = (((ic * out_channels + b) * kernel_d + kd)
                                        * kernel_h + kh) * kernel_w + kw;
                        sum += input[in_idx] * weight[weight_idx];
                    }
                }
            }
        }
    }
    if (bias != 0) {
        sum += bias[b * out_channels + ic];
    }
    output[idx] = sum;
}
