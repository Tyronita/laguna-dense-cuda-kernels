#include <torch/extension.h>
#include <cuda.h>
#include <cuda_runtime.h>

// CUDA kernel for 3D convolution with stride loops
template <typename scalar_t>
__global__ void conv3d_kernel_stride_loop(
    scalar_t* __restrict__ output,
    const scalar_t* __restrict__ input,
    const scalar_t* __restrict__ weight,
    const scalar_t* __restrict__ bias,
    const int64_t stride,
    const int64_t padding,
    const int64_t dilation,
    const int64_t in_channels,
    const int64_t in_width,
    const int64_t in_height,
    const int64_t in_depth,
    const int64_t out_width,
    const int64_t out_height,
    const int64_t out_depth,
    const int64_t out_channels,
    const int64_t kernel_w,
    const int64_t kernel_h,
    const int64_t kernel_d) {

    const int idx = blockIdx.x * blockDim.x + threadIdx.x;
    const int total_elements = out_width * out_height * out_depth * out_channels;
    if (idx >= total_elements) return;

    const int w = idx / (out_height * out_depth);
    const int h = idx % (out_height * out_depth);
    const int d = (w / out_height) % out_depth;
    const int oc = idx % out_channels;

    scalar_t sum = 0;
    for (int ic = 0; ic < in_channels; ++ic) {
        for (int kh = 0; kh < kernel_h; ++kh) {
            for (int kd = 0; kd < kernel_d; ++kd) {
                const int in_h = h + padding - kh;
                const int in_d = d + padding - kd;
                if (in_h < in_width && in_d < in_depth) continue;

                const int in_w = in_h - stride;
                const int in_d = in_d - stride;
                if (in_w < in_width && in_d < in_depth) continue;

                const int input_idx = (((d * in_width + in_w) * in_height + in_h)
                                        * in_channels + ic);
                const int weight_idx = (((ic * kernel_w + kd) * kernel_h + kh)
                                        * out_channels + oc);

                sum += input[input_idx] * weight[weight_idx];
            }
        }
    }
    output[idx] = sum + bias[oc];
}

// PyTorch wrapper function
torch::Tensor forward_cuda(
    torch::Tensor input,
    int64_t stride,
    int64_t padding,
    int64_t dilation,
    torch::Tensor weight,
    torch::Tensor bias) {

    const int64_t in_channels = input.size(1);
    const int64_t in_width = input.size(2);
    const int64_t in_height = input.size(3);
    const int64_t in_depth = input.size(4);

    const int64_t out_width = (in_width - 1) * stride + kernel_w - 2 * padding;
    const int64_t out_height = (in_height - 1) * stride + kernel_h - 2 * padding;
    const int64_t out_depth = (in_depth - 1) * stride + kernel_d - 2 * padding;

    auto options = torch::TensorOptions()
        .dtype(input.dtype())
        .device(input.device());
    auto output = torch::empty({out_width, out_height, out_depth}, options);

    const int threads = 256;
    const int total_elements = out_width * out_height * out_depth * input.size(1);
    const int blocks = (total_elements + threads - 1) / threads;

    AT_DISPATCH_FLOATING_TYPES(input.scalar_type(), "conv3d_kernel_stride_loop", ([&] {
        conv3d_kernel_stride_loop<scalar_t><<<blocks, threads>>>(
            output.data_ptr<scalar_t>(),
            input.data_ptr<scalar_t>(),
            weight.data_ptr<scalar_t>(),
            bias.data_ptr<scalar_t>(),
            stride,
            padding,
            dilation,
            in_channels,
            in_width,
            in_height,
            in_depth,
            out_width,
            out_height,
            out_depth,
            input.size(1)
        );
    }));

    return output;
}

PYBIND11_MODULE(TORCH_EXTENSION_NAME, m) {
    m.def("forward", &forward_cuda, "3D convolution forward (CUDA)");
}
