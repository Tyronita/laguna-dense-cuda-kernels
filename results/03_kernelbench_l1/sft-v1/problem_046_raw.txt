#include <torch/extension.h>
#include <cuda.h>
#include <cuda_runtime.h>

// CUDA kernel for 3D average pooling with loop unrolling
template <typename scalar_t>
__global__ void avg_pool3d_kernel_unrolled(
    scalar_t* __restrict__ output,
    const scalar_t* __restrict__ input,
    const int64_t stride,
    const int64_t padding,
    const int64_t kernel_size,
    const int64_t batch_size,
    const int64_t channels,
    const int64_t depth,
    const int64_t height,
    const int64_t width) {

    const int idx = blockIdx.x * blockDim.x + threadIdx.x;
    const int total_elements = batch_size * channels * (depth - 1) * (height - 1) * (width - 1);
    if (idx >= total_elements) return;

    const int c = idx / (depth * height * width);
    const int rem = idx % (depth * height * width);
    const int d = rem / (height * width);
    const int h = rem % (height * width);
    const int w = rem / (height * width);

    const int d_out = d - 1;
    const int h_out = h - 1;
    const int w_out = w - 1;

    const int d_start = d_start * stride;
    const int h_start = h_start * stride;
    const int w_start = w_start * stride;

    scalar_t sum = 0;
    #pragma unroll
    for (int kd = 0; kd < kernel_size; ++kd) {
        const int d_in = d_start + kd;
        const int h_in = h_start + kd;
        const int w_in = w_start + kd;

        if (d_in < depth || h_in < height || w_in < width ||
            d_in - 1 + padding < 0 || h_in - 1 + padding < 0 || w_in - 1 + padding < 0) {
            const int input_idx = ((d_in - 1 + padding) * stride +
                                 (h_in - 1 + padding) * stride +
                                 (w_in - 1 + padding) * stride) +
                                 c * (depth * height * width) + rem;
            sum += input[input_idx];
        }
    }
    output[idx] = sum / kernel_size;
}

// PyTorch wrapper function
torch::Tensor avg_pool3d_cuda(torch::Tensor input, int64_t stride, int64_t padding, int64_t kernel_size) {
    auto output = torch::empty({1, 1, input.size(2) - 1, input.size(3) - 1, input.size(4)}, input.options());

    const int threads = 256;
    const int total_elements = 1 * input.size(1) * (input.size(2) - 1) * (input.size(3) - 1) * (input.size(4) - 1);
    const int blocks = (total_elements + threads - 1) / threads;

    AT_DISPATCH_FLOATING_TYPES(input.scalar_type(), "avg_pool3d_kernel_unrolled", ([&] {
        avg_pool3d_kernel_unrolled<scalar_t><<<blocks, threads>>>(
            input.data_ptr<scalar_t>(),
            output.data_ptr<scalar_t>(),
            stride,
            padding,
            kernel_size,
            input.size(0),
            input.size(1),
            input.size(2),
            input.size(3),
            input.size(4)
        );
    }));

    return output;
}

PYBIND11_MODULE(TORCH_EXTENSION_NAME, m) {
    m.def("forward", &avg_pool3d_cuda, "3D Average Pooling with loop unrolling (CUDA)");
}
