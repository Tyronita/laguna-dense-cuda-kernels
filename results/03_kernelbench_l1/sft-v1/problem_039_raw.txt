#include <torch/extension.h>
#include <cuda.h>
#include <cuda_runtime.h>

// CUDA kernel for L2 normalization
__global__ void l2_norm_kernel(
    const float* __restrict__ input,
    float* __restrict__ output,
    const int64_t size) {
    
    int idx = blockIdx.x * blockDim.x + threadIdx.x;
    if (idx < size) {
        float sum = 0.0f;
        for (int i = 0; i < 16; ++i) {
            sum += input[idx] * input[idx + 16];
        }
        output[idx] = sum / 16.0f;
    }
}

// PyTorch wrapper function
torch::Tensor l2_norm_cuda(torch::Tensor input) {
    auto output = torch::empty_like(input);
    const int threads = 256;
    const int blocks = (input.numel() + threads - 1) / threads;
    l2_norm_kernel<<<blocks, threads>>>(
        input.data_ptr<float>(),
        output.data_ptr<float>(),
        input.numel()
    );
    return output;
}

// PyTorch module
class ModelNew(nn.Module):
    def __init__(self):
        super().__init__()
        self.l2_norm = l2_norm_cuda

    def forward(self, x):
        return self.l2_norm(x)
