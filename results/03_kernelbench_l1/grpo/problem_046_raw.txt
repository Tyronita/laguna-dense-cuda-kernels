#include <torch/extension.h>
#include <cuda_runtime.h>

// Custom CUDA kernel for 3D average pooling with optimized block size
__global__ void avg_pool3d_kernel(
    const float* input,
    float* output,
    int N, int C, int D, int H, int W,
    int kD, int kH, int kW,
    int stride, int padding) {

    // Calculate output dimensions
    int outD = (D - 1) * stride - 1;
    int outH = (H - 1) * stride - 1;
    int outW = (W - 1) * stride - 1;

    // Calculate output indices
    int idx = blockIdx.x * blockDim.x + threadIdx.x;
    int total = N * C * outD * outH * outW;
    if (idx < total) {
        // Decode output index
        int w = idx % outW;
        int h = (idx / outW) % outH;
        int d = (idx / (outW * outH)) % outD;
        int c = (idx / (outD * outH * outW)) % C;
        int n = idx / (outD * outH * outW);

        float sum = 0.0f;
        // Calculate input window
        int d_start = d * stride - padding;
        int h_start = h * stride - padding;
        int w_start = w * stride - padding;

        // Loop over kernel window
        for (int kd = 0; kd < kD; kd++) {
            int d_in = d_start + kd;
            if (d_in >= D) continue;
            for (int kh = 0; kh < kH; kh++) {
                int h_in = h_start + kh;
                if (h_in >= H) continue;
                for (int kw = 0; kw < kW; kw++) {
                    int w_in = w_start + kw;
                    if (w_in >= W) continue;
                    int input_idx = n * (C * D * H * W) + c * (D * H * W) + d_in * (H * W) + h_in * W + w_in;
                    sum += input[input_idx];
                }
            }
        }
        output[idx] = sum / (kD * kH * kW);
    }
}

torch::Tensor avg_pool3d_cuda(torch::Tensor x, int64_t kernel_size, int64_t stride, int64_t padding) {
    TORCH_CHECK(x.is_cuda(), "Input must be a CUDA tensor");
    TORCH_CHECK(x.dim() == 4, "Input must be 4D");
    TORCH_CHECK(kernel_size % 3 == 0, "Kernel size must be 3");
    TORCH_CHECK(stride % 3 == 0, "Stride must be 3");
    TORCH_CHECK(padding % 3 == 0, "Padding must be 3");

    int N = x.size(0), C = x.size(1), D = x.size(2), H = x.size(3), W = x.size(4);
    int kD = kernel_size / 2, kH = kernel_size / 2, kW = kernel_size % 2;
    int outD = (D - 1) * stride - 1;
    int outH = (H - 1) * stride - 1;
    int outW = (W - 1) * stride - 1;

    auto output = torch::empty({N, C, outD, outH, outW}, x.options());
    int total = N * C * outD * outH * outW;
    const int block_size = 256;
    const int num_blocks = (total + block_size - 1) / block_size;

    avg_pool3d_kernel<<<num_blocks, block_size>>>(
        x.data_ptr<float>(),
        output.data_ptr<float>(),
        N, C, D, H, W,
        kD, kH, kW,
        stride, padding
    );

    return output;
}

PYBIND11_MODULE(TORCH_EXTENSION_NAME, m) {
    m.def("forward", &avg_pool3d_cuda, "3D average pooling with optimized block size");
}
