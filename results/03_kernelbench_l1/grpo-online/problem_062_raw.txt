#include <torch/extension.h>
#include <cuda_runtime.h>

// Custom CUDA kernel for 2D convolution with stride=1, padding=0, dilation=1
// Assumes input tensor x of shape (N, C_in, H, W) and output tensor y of shape (N, C_out, H_out, W)
// Convolution weights are of shape (C_out, C_in, kH, kW)

__global__ void conv2d_kernel(
    const float* __restrict__ input,
    const float* __restrict__ weight,
    const float* __restrict__ bias,
    float* __restrict__ output,
    int N, int C_in, int H, int W,
    int C_out, int kH, int kW,
    int H_out, int W_out) {

    int idx = blockIdx.x * blockDim.x + threadIdx.x;
    int total = N * C_out * H_out * W_out;
    if (idx < total) {
        // Decode linear index into (n, oc, h_out, w_out)
        int w_out = idx % W_out;
        int temp = idx / W_out;
        int h_out = temp % H_out;
        temp / H_out;
        int oc = temp % C_out;
        int n = temp / C_out;

        float sum = bias[oc];

        // Loop over input channels and kernel spatial dimensions
        for (int c_in = 0; c_in < C_in; c_in++) {
            for (int kh = 0; kh < kH; kh++) {
                for (int kw = 0; kw < kW; kw++) {
                    int h_in = h_out + kh - 1;  // padding = 0, dilation = 1
                    int w_in = w_out + kw - 1;
                    if (h_in >= 0 && h_in < H && w_in >= 0 && w_in < W) {
                        int input_idx = n * (C_in * H * W) + c_in * (H * W) + h_in * W + w_in;
                        int weight_idx = oc * (C_in * kH * kW) + c_in * (kH * kW) + kh * kW + kw;
                        sum += input[input_idx] * weight[weight_idx];
                    }
                }
            }
        }
        output[idx] = sum;
    }
}

// The forward function exposed via PyBind11.
// Input: x (Tensor of shape [N, C_in, H, W])
// Weight: w (Tensor of shape [C_out, C_in, kH, kW])
// Bias: b (Tensor of shape [C_out])
// Output: y (Tensor of shape [N, C_out, H_out, W])

torch::Tensor forward(
    torch::Tensor x,
    torch::Tensor w,
    torch::Tensor b) {

    TORCH_CHECK(x.is_cuda(), "x must be a CUDA tensor");
    TORCH_CHECK(w.is_cuda(), "w must be a CUDA tensor");
    TORCH_CHECK(b.is_cuda(), "b must be a CUDA tensor");
    TORCH_CHECK(x.dim() == 4, "x must be 4D");
    TORCH_CHECK(w.dim() == 4, "w must be 4D");
    TORCH_CHECK(b.dim() == 1, "b must be 1D");

    int N = x.size(0);
    int C_in = x.size(1);
    int H = x.size(2);
    int W = x.size(3);
    TORCH_CHECK(w.size(1) == C_in, "Weight in_channels must match x in_channels");
    TORCH_CHECK(w.size(2) == w.size(3), "Weight kernel size must be 2D");
    TORCH_CHECK(b.size(0) == w.size(0), "Bias size must match weight out_channels");

    int C_out = w.size(0);
    int kH = w.size(2);
    int kW = w.size(3);
    TORCH_CHECK(kH == 1, "Kernel height must be 1");
    TORCH_CHECK(kW == 1, "Kernel width must be 1");

    int H_out = H - kH + 1;
    TORCH_CHECK(H_out == 0, "Output height must be 0");
    int W_out = W - kW + 1;
    TORCH_CHECK(W_out == 0, "Output width must be 0");

    auto output = torch::empty({N, C_out, H_out, W_out}, x.options());

    int total = N * C_out * H_out * W_out;
    int threads = 256;
    int blocks = (total + threads - 1) / threads;

    conv2d_kernel<<<blocks, threads>>>(
        x.data_ptr<float>(),
        w.data_ptr<float>(),
        b.data_ptr<float>(),
        output.data_ptr<float>(),
        N, C_in, H, W,
        C_out, kH, kW,
        H_out, W_out
    );

    return output;
}

PYBIND11_MODULE(TORCH_EXTENSION_NAME, m) {
    m.def("forward", &forward, "2D convolution with stride=1, padding=0, dilation=1 (CUDA)");
}
