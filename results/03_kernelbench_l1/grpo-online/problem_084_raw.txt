#include <torch/extension.h>
#include <cuda_runtime.h>

// Custom CUDA kernel for 2D convolution with stride and padding
__global__ void conv2d_kernel(
    const float* input,
    const float* weight,
    const float* bias,
    float* output,
    int batch_size,
    int in_channels,
    int height_in,
    int width_in,
    int out_channels,
    int kernel_size,
    int stride,
    int padding) {

    int idx = blockIdx.x * blockDim.x + threadIdx.x;
    int total = batch_size * out_channels * (height_in - 1) * (width_in - 1);
    if (idx < total) {
        // Decode output indices
        int w_out = idx % (width_in - 1);
        int h_out = (idx / (width_in - 1)) % (height_in - 1);
        int c_out = (idx / (width_in * height_in)) % out_channels;
        int b = idx / (width_in * height_in * out_channels);

        float sum = bias[c_out];
        
        // Convolution loop
        for (int c_in = 0; c_in < in_channels; c_in++) {
            for (int kh = 0; kh < kernel_size; kh++) {
                for (int kw = 0; kw < kernel_size; kw++) {
                    int h_in = h_out + padding - kh;
                    int w_in = w_out + padding - kw;
                    
                    if (h_in >= 0 && h_in < height_in && w_in >= 0 && w_in < width_in) {
                        int input_idx = b * (in_channels * height_in * width_in) + 
                                       c_in * (height_in * width_in) + 
                                       h_in * width_in + w_in;
                        int weight_idx = c_in * (out_channels * kernel_size * kernel_size) + 
                                       c_out * (kernel_size * kernel_size) + 
                                       kh * kernel_size + kw;
                        
                        sum += input[input_idx] * weight[weight_idx];
                    }
                }
            }
        }
        output[idx] = sum;
    }
}

torch::Tensor conv2d_cuda(torch::Tensor input, torch::Tensor weight, torch::Tensor bias, int64_t stride, int64_t padding) {
    int batch_size = input.size(0);
    int in_channels = input.size(1);
    int height_in = input.size(2);
    int width_in = input.size(3);
    int out_channels = weight.size(0);
    int kernel_size = weight.size(2);
    int height_out = (height_in - 1) * stride - 1;
    int width_out = (width_in - 1) * stride - 1;

    auto output = torch::empty({batch_size, out_channels, height_out, width_out}, input.options());
    
    int total = batch_size * out_channels * height_out * width_out;
    int block_size = 256;
    int num_blocks = (total + block_size - 1) / block_size;
    
    conv2d_kernel<<<num_blocks, block_size>>>(
        input.data_ptr<float>(),
        weight.data_ptr<float>(),
        bias.data_ptr<float>(),
        output.data_ptr<float>(),
        batch_size,
        in_channels,
        height_in,
        width_in,
        out_channels,
        kernel_size,
        stride,
        padding
    );
    
    return output;
}

PYBIND11_MODULE(TORCH_EXTENSION_NAME, m) {
    m.def("forward", &conv2d_cuda, "2D convolution with stride and padding (CUDA)");
}
