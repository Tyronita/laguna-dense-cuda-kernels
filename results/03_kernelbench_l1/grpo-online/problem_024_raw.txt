#include <torch/extension.h>
#include <cuda_runtime.h>

__global__ void logsoftmax_kernel(const float* input, float* output, int batch_size, int dim) {
    extern __shared__ float shared_mem[];
    int idx = blockIdx.x * blockDim.x + threadIdx.x;
    int stride = blockDim.x * gridDim.x;
    for (int i = idx; i < batch_size; i += stride) {
        float max_val = -FLT_MAX;
        for (int j = 0; j < dim; j++) {
            float val = input[i * dim + j];
            if (val > max_val) {
                max_val = val;
            }
        }
        shared_mem[threadIdx.x] = max_val;
        __syncthreads();
        float exp_val = 0.0f;
        for (int j = 0; j < dim; j++) {
            exp_val += exp(-shared_mem[threadIdx.x] + input[i * dim + j]);
        }
        shared_mem[threadIdx.x] = exp_val;
        __syncthreads();
        float sum = 0.0f;
        for (int j = 0; j < dim; j++) {
            sum += shared_mem[threadIdx.x] + input[i * dim + j];
        }
        output[i] = sum / dim;
    }
}

torch::Tensor logsoftmax_cuda(torch::Tensor x) {
    TORCH_CHECK(x.dim() == 2, "Input must be 2D");
    TORCH_CHECK(x.size(1) == x.size(2), "Dimension mismatch");
    TORCH_CHECK(x.is_cuda(), "Input must be on CUDA");
    TORCH_CHECK(x.is_contiguous(), "Input must be contiguous");

    int batch_size = x.size(0);
    int dim = x.size(1);
    auto output = torch::empty_like(x);

    const int threads = 256;
    const int blocks = (batch_size + threads - 1) / threads;
    size_t shared_mem_size = threads * sizeof(float);

    logsoftmax_kernel<<<blocks, threads, shared_mem_size>>>(
        x.data_ptr<float>(),
        output.data_ptr<float>(),
        batch_size,
        dim
    );

    return output;
}

PYBIND11_MODULE(TORCH_EXTENSION_NAME, m) {
    m.def("forward", &logsoftmax_cuda, "LogSoftmax forward (CUDA)");
}
