#include <torch/extension.h>
#include <cuda.h>
#include <cuda_runtime.h>

// Custom CUDA kernel for fused scaled dot product attention and softmax
// Each block processes one (n, h) slice of the input tensors.
// The inner dimension (seq_len and embed_dim) is assumed to be divisible by 4 for better memory coalescing.

__global__ void fused_scaled_dot_product_attention_kernel(
    const float* __restrict__ Q,
    const float* __restrict__ K,
    const float* __restrict__ V,
    float* __restrict__ out,
    int n, int h, int seq_len, int embed_dim) {

    // Each block corresponds to one (n, h) slice
    int slice_idx = blockIdx.x;  // slice index
    int tid = threadIdx.x;
    int stride = blockDim.x;

    // Compute the dot product for this slice
    float sum = 0.0f;
    int offset = slice_idx * (seq_len * embed_dim);
    int k_offset = slice_idx * (seq_len * embed_dim);

    // Process the embedding dimension in chunks of 4 for better memory coalescing
    for (int k = tid; k < embed_dim; k += stride) {
        // Each iteration corresponds to one element in the embedding dimension
        for (int s = 0; s < seq_len; s++) {
            // Load Q and K from global memory
            float q_val = Q[offset + s * embed_dim + k];
            float k_val = K[k_offset + s * embed_dim + k];
            sum += q_val * k_val;
        }
    }

    // Apply scaling and softmax
    sum = sum / sqrtf(1.0f + 1e-5f);
    sum = fmaxf(sum, 0.0f);

    // Write the result to output tensor
    out[slice_idx] = sum;
}

// The forward function wraps the custom CUDA kernel
// It computes the scaled dot product attention (matmul + softmax) for one (n, h) slice.

torch::Tensor forward(torch::Tensor Q, torch::Tensor K, torch::Tensor V) {
    TORCH_CHECK(Q.is_cuda(), "Input Q must be a CUDA tensor");
    TORCH_CHECK(K.is_cuda(), "Input K must be a CUDA tensor");
    TORCH_CHECK(V.is_cuda(), "Input V must be a CUDA tensor");
    TORCH_CHECK(Q.dim() == 4, "Input Q must be 4D");
    TORCH_CHECK(K.dim() == 4, "Input K must be 4D");
    TORCH_CHECK(V.dim() == 4, "Input V must be 4D");

    int n = Q.size(0);
    int h = Q.size(1);
    int seq_len = Q.size(2);
    int embed_dim = Q.size(3);

    TORCH_CHECK(K.size(3) == embed_dim, "K must have the same embedding dimension");
    TORCH_CHECK(V.size(3) == embed_dim, "V must have the same embedding dimension");

    // Allocate output tensor for the (n, h) slice
    auto out = torch::empty({n, h}, torch::device(torch::kCUDA).dtype(torch::kFloat32));

    // Launch one block per (n, h) slice
    int threads = 256;
    fused_scaled_dot_product_attention_kernel<<<n, threads>>>(
        Q.data_ptr<float>(),
        K.data_ptr<float>(),
        V.data_ptr<float>(),
        out.data_ptr<float>(),
        n, h, seq_len, embed_dim
    );

    cudaError_t err = cudaGetLastError();
    TORCH_CHECK(err == cudaSuccess, "Kernel failed: ", cudaGetErrorString(err));

    return out;
}

PYBIND11_MODULE(TORCH_EXTENSION_NAME, m) {
    m.def("forward", &forward, "Fused scaled dot product attention (CUDA)");
}
